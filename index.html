document.addEventListener('DOMContentLoaded', () => { const buttons = document.querySelectorAll('.switch-btn'); const layers = document.querySelectorAll('.light-layer'); const viewport = document.querySelector('.simulator-viewport'); // 取得畫面外框來套用濾鏡 function adjustDynamicVisuals() { const activeLayers = document.querySelectorAll('.light-layer.active'); const count = activeLayers.length; let targetOpacity = 1.0; let brightnessFilter = 1.0; let contrastFilter = 1.0; // 依據開燈數量，動態對齊真實全開照片 (zen-full.jpg) 的視覺質感 if (count <= 1) { targetOpacity = 1.0; // 單燈：維持 100% 真實扎實度 brightnessFilter = 1.0; contrastFilter = 1.0; } else if (count === 2) { targetOpacity = 0.85; brightnessFilter = 1.0; contrastFilter = 1.0; } else if (count === 3) { targetOpacity = 0.75; brightnessFilter = 0.95; // 開始微調整體亮度 contrastFilter = 1.05; // 稍微拉高對比，把被吃掉的線條細節逼出來 } else { // 四盞燈以上（邁向全開）：強效校正，模擬真實照片的細節保留 // 燈開越多，透明度與亮度上限適度壓低，對比度提高 targetOpacity = Math.max(0.50, 1.0 - (count * 0.07)); brightnessFilter = Math.max(0.85, 1.0 - (count * 0.02)); // 防止高光區域過曝死白 contrastFilter = Math.min(1.20, 1.0 + (count * 0.03)); // 強化天花板與暗部細節 } // 1. 套用各圖層的動態透明度 layers.forEach(layer => { if (layer.classList.contains('active')) { layer.style.opacity = targetOpacity; } else { layer.style.opacity = 0; } }); // 2. 直接對整個模擬畫面套用濾鏡校正，讓全開時的細節完美浮現 if (count > 2) { viewport.style.filter = `brightness(${brightnessFilter}) contrast(${contrastFilter})`; } else { viewport.style.filter = 'none'; // 燈少時維持原汁原味 } } buttons.forEach(button => { button.addEventListener('click', () => { const targetLayerId = button.getAttribute('data-target'); const targetLayer = document.getElementById(targetLayerId); button.classList.toggle('on'); if (button.classList.contains('on')) { if (targetLayer) targetLayer.classList.add('active'); } else { if (targetLayer) targetLayer.classList.remove('active'); } // 觸發校正機制 adjustDynamicVisuals(); }); }); });
